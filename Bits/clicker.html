What is the output of the following program if I call it with: 
UNIX> echo 0xa5 | ./a.out a b c d e f g h 


#include <vector>
#include <iostream>
using namespace std;

int main(int argc, char **argv)
{
  vector <string> args;
  int set;
  int i;

  for (i = 1; i < argc; i++) args.push_back(argv[i]);
  cin >> hex >> set;

  for (i = 0; i < args.size(); i++) {
     if (set & (1 << i)) cout << args[i];
  }
  cout << endl;
  return 0;
}

Answer: 0xa5 is 1010 0101 in binary, so this will print entries 0, 2, 5 and 7 from the vector:
