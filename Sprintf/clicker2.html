What is the output of the following program? (src/click2.cpp) 

#include <cstdio>
#include <iostream>
using namespace std;

int main()
{
  int x, y;
  string s;
 
  x = 5;
  y = 10;
  s = "15 Fred";

  sscanf(s.c_str(), "%d %d", &x, &y);
  cout << x << "-" << y << "-";

  s = "Fred 20";

  sscanf(s.c_str(), "%d %d", &x, &y);
  cout << x << "-" << y << endl;

  return 0;
}

The first sscanf() correctly reads 15 into x, but fails reading y, because "Fred" is not an integer. So y remains at 10. The first sscanf() fails reading x, and the sscanf() call exits at that point, so y remains unchanged. The answer is: 

UNIX> make bin/click2
g++ -std=c++98 -Wall -Wextra -o bin/click2 src/click2.cpp
UNIX> bin/click2
15-10-15-10
UNIX>
